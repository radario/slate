    limit query integer false Specifies the number of items to return
    offset query integer false Specifies the number of items to skip
    groupId query string false Id of event group
    onlyWithOrderCreationAvailableViaApi query boolean false Should specify true to return events with permission to create

            "placeAddress": "string",
            "companyId": "int",
            "companyTitle": "string",
            "minPrice": "int",
            "maxPrice": "int",
            "cityId": "int",
            "cityName": "string",
            "placeSchemeId": "int",
            "groupId": "int",
            "currency": "string",
            "ticketCount": "int",
            "gmtOffset": "int"

    gmtOffset GMT offset of city where event take place
